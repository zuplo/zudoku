import { Secret } from "zudoku/ui/Secret";
import { Value } from "zudoku/ui/Value";

The `Secret` component can be used to display secrets, typically API keys, in
your Zudoku application.

## Import

```tsx
import { Secret } from "zudoku/ui/Secret";
```

## Usage

```tsx
<Secret
  secret="pkcd_this-is-a-never-01K1DASDXXQDA9S6GKASKYAFSDH"
  status="active"
  preview={5}
  onCopy={(secret) => {
    console.log("Copied secret:", secret);
  }}
  onReveal={(revealed) => {
    console.log("Revealed:", revealed);
  }}
/>
```
